let hackHr =0;		//Set hours
let hackMin = 5; 	//Set minutes
let hackSec = 18;	//Set seconds
let hackRHr = 0;	//Set reset hour
let hackRMin = 0;	//Set reset min
let hackRSec = 9;	//Set reset sec

handleCustomLoop = () => {
	let hackTime = document.getElementsByClassName("video-stream html5-main-video")[0].currentTime;
	if (hackTime > hackHr*3600 + hackMin*60 + hackSec) {
		document.getElementsByClassName("video-stream html5-main-video")[0].currentTime = hackRHr*3600+hackRMin*60+hackSec;
	}
}
document.getElementsByClassName("video-stream html5-main-video")[0].addEventListener("timeupdate", handleCustomLoop);
